Applicatie met database

under Software

Hieronder de code om de database aan te maken.

create schema gip_taak_08_SegersWout;
 use gip_taak_08_SegersWout;
 
 create table UserAccount(
	id int not null auto_increment primary key,
    naam text not null,
    email text not null, 
    age int
    );
 
 create table Song(
	id int not null auto_increment primary key,
	user_id int not null,
    song_naam text not null, 
	artiest text,
    likes int
);
 create table Playlist(
	id int not null auto_increment primary key,
	user_id int ,
    song_id int , 
    playlist_naam text 
);

SELECT UserAccount.naam, UserAccount.email, UserAccount.age,
Song.song_naam, Song.artiest, Song.likes, Playlist.playlist_naam
FROM UserAccount JOIN Song 
ON UserAccount.id = Song.user_id
INNER JOIN Playlist ON(Song.id = Playlist.id);
